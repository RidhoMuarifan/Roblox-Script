    local TabContainer = Instance.new("ScrollingFrame")

        TabContainer.CanvasSize = UDim2.new(0, 0, 0, ContentLayout.AbsoluteContentSize.Y + 10)

    TabBtn.Size = UDim2.new(0, 120, 0, 35)
    TabBtn.BackgroundColor3 = Color3.fromRGB(24, 24, 30)
    TabBtn.BackgroundTransparency = 1
    TabBtn.Text = "  " .. name
    TabBtn.TextColor3 = Color3.fromRGB(150, 150, 150)
    TabBtn.Font = Enum.Font.GothamBold
    TabBtn.TextSize = 12
    TabBtn.TextXAlignment = Enum.TextXAlignment.Left
    TabBtn.Parent = Sidebar
    Instance.new("UICorner", TabBtn).CornerRadius = UDim.new(0, 4)
    
    Tabs[name] = TabContainer
    TabButtons[name] = TabBtn
    
    TabBtn.MouseButton1Click:Connect(function()
        for tName, container in pairs(Tabs) do
            container.Visible = (tName == name)
            TweenService:Create(TabButtons[tName], TweenInfo.new(0.2), {
                BackgroundTransparency = (tName == name) and 0 or 1,
                BackgroundColor3 = (tName == name) and Color3.fromRGB(35, 35, 45) or Color3.fromRGB(24, 24, 30),
                TextColor3 = (tName == name) and Color3.fromRGB(255, 255, 255) or Color3.fromRGB(150, 150, 150)
            }):Play()
        end
    end)
    
    return TabContainer
end

local function CleanupScript()
    if not ScriptRunning then return end
    ScriptRunning = false
    Config.AutoEat = false
    Config.AutoSell = false
    Config.RandomTeleportEnabled = false
    Config.Upgrade_MaxSize = false
    Config.Upgrade_Multiplier = false
    Config.Upgrade_Speed = false
    Config.Upgrade_EatSpeed = false
    
    for _, conn in pairs(Connections) do
        if conn then pcall(function() conn:Disconnect() end) end
    end

    pcall(function() FloatingGui:Destroy() end)

        _G.ETW_DedicatedHubCleanup = nil
    end
end

-- CORE GAME REQUISITES
local GrabRemotesCache, EatRemotesCache = {}, {}
local lastRemotesRefresh, autoEatPausedUntil = 0, 0
local AutoEatPauseInterval, AutoEatPauseDuration = 30, 1
local nextAutoEatPauseAt = tick() + AutoEatPauseInterval
local MapPartsCache, lastMapPartsRefresh = {}, 0
local RandomTeleportInterval = 3
local RandomGenerator = Random.new()
local BlockedTeleportNames = {"sell", "cash", "shop", "trade", "claim", "market"}
local BlockedActionRemoteNames = {"sell", "cash", "shop", "trade", "claim", "market", "progress", "bar", "fill", "money", "coin", "reward", "collect"}

local function HasNameInPath(instance, names)
    local current = instance
    while current and current ~= workspace do
        local currentName = string.lower(current.Name)
        for _, name in ipairs(names) do
            if string.find(currentName, name, 1, true) then return true end
        end
        current = current.Parent
    end
    return false
end

local function RefreshMapParts()
    MapPartsCache = {}
    lastMapPartsRefresh = tick()
    local mapRoot = workspace:FindFirstChild("Map") or workspace
    local character = LocalPlayer.Character
    for _, part in pairs(mapRoot:GetDescendants()) do
        if part:IsA("BasePart") and part.CanCollide and part.Transparency < 1 then
            if not (character and part:IsDescendantOf(character)) and part.Size.X >= 6 and part.Size.Z >= 6 and not HasNameInPath(part, BlockedTeleportNames) then
                table.insert(MapPartsCache, part)
            end
        end
    end
end

local function TeleportToRandomMapSpot()
    local character = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
    local root = character:FindFirstChild("HumanoidRootPart")
    if not root then return end
    if #MapPartsCache == 0 or (tick() - lastMapPartsRefresh) > 5 then RefreshMapParts() end
    if #MapPartsCache == 0 then return end
    local targetPart = MapPartsCache[RandomGenerator:NextInteger(1, #MapPartsCache)]
    if not targetPart or not targetPart.Parent then RefreshMapParts() return end
    
    local safeX = math.max(targetPart.Size.X / 2 - 2, 0)
    local safeZ = math.max(targetPart.Size.Z / 2 - 2, 0)
    root.CFrame = targetPart.CFrame * CFrame.new(RandomGenerator:NextNumber(-safeX, safeX), targetPart.Size.Y / 2 + 5, RandomGenerator:NextNumber(-safeZ, safeZ))
end

local function RefreshRemotes()
    GrabRemotesCache, EatRemotesCache = {}, {}
    lastRemotesRefresh = tick()
    local function scan(root)
        for _, v in pairs(root:GetDescendants()) do
            if v:IsA("RemoteEvent") and not HasNameInPath(v, BlockedActionRemoteNames) then
                local n = string.lower(v.Name)
                if n == "grab" then table.insert(GrabRemotesCache, v)
                elseif n == "eat" then table.insert(EatRemotesCache, v) end
            end
        end
    end
    pcall(function() scan(ReplicatedStorage) end)
    pcall(function() if LocalPlayer.Character then scan(LocalPlayer.Character) end end)
end

local GetSizeData
local function CanSellNow()
    if not GetSizeData then return false end
    local cSize, mSize = GetSizeData()
    return cSize > 0 and mSize > 0 and cSize >= mSize and mSize ~= math.huge
end

local function GetSellRemote()
    return LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("Events") and LocalPlayer.Character.Events:FindFirstChild("Sell")
end

local function FireSellRemotes(forceSell)
    if not ScriptRunning then return end
    if not forceSell and (not Config.AutoSell or not CanSellNow()) then return end
    local sellRemote = GetSellRemote()
    if sellRemote then pcall(function() sellRemote:FireServer() end) end
end

table.insert(Connections, RunService.Heartbeat:Connect(function()
    if not ScriptRunning or not Config.AutoEat then return end
    if tick() < autoEatPausedUntil or LocalPlayer.Character and LocalPlayer.Character:FindFirstChildOfClass("Humanoid") and (LocalPlayer.Character:FindFirstChildOfClass("Humanoid"):GetState() == Enum.HumanoidStateType.Freefall) then return end
    
    if tick() >= nextAutoEatPauseAt then
        autoEatPausedUntil = tick() + AutoEatPauseDuration
        nextAutoEatPauseAt = tick() + AutoEatPauseInterval + AutoEatPauseDuration
        return
    end
    
    if #GrabRemotesCache == 0 or #EatRemotesCache == 0 or (tick() - lastRemotesRefresh) > 5 then RefreshRemotes() end
    for _, r in ipairs(GrabRemotesCache) do if r.Parent then pcall(function() r:FireServer() end) end end
    for _, r in ipairs(EatRemotesCache) do if r.Parent then pcall(function() r:FireServer() end) end end
end))

task.spawn(function()
    while ScriptRunning do
        task.wait(RandomTeleportInterval)
        if Config.RandomTeleportEnabled then TeleportToRandomMapSpot() end
    end
end)

local MaxSizeGrowthFunction
pcall(function() MaxSizeGrowthFunction = require(ReplicatedStorage:WaitForChild("ItemInfo")).Upgrades.MaxSize.growthFunction end)
local function ApplySizeGrowth(value)
    if MaxSizeGrowthFunction then
        local success, result = pcall(MaxSizeGrowthFunction, value)
        if success and type(result) == "number" then return result end
    end
    return value
end

GetSizeData = function()
    local currentSize, maxSize = 0, math.huge
    if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("Size") then
        currentSize = ApplySizeGrowth(LocalPlayer.Character.Size.Value)
    end
    local upg = LocalPlayer:FindFirstChild("Upgrades") or LocalPlayer:FindFirstChild("upgrades") or LocalPlayer:FindFirstChild("Upgrade")
    if upg and upg:FindFirstChild("MaxSize") then
        maxSize = ApplySizeGrowth(upg.MaxSize.Value)
    end
    return currentSize, maxSize
end
